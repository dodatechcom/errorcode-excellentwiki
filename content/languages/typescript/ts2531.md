# TS2531 — Object is possibly 'null'

TS2531 occurs when you access a property or call a method on an object that could be `null`. This is enforced when `strictNullChecks` is enabled in your tsconfig, which is the default in modern TypeScript projects.

## Common Causes

```typescript
// Cause 1: DOM query that might return null
const el = document.getElementById("app");
el.innerHTML = "Hello"; // TS2531: Object is possibly 'null'

// Cause 2: Function parameter that accepts null
function process(value: string | null) {
  console.log(value.length); // TS2531
}

// Cause 3: Array.find() returns undefined
const items = [1, 2, 3];
const found = items.find(x => x === 99);
console.log(found.toFixed()); // TS2531 (find returns T | undefined)

// Cause 4: Optional property access without null check
interface Config {
  host?: string;
}
function getHost(config: Config) {
  return config.host.toUpperCase(); // TS2531
}
```

## Solutions

### Fix 1: Add a null check before accessing properties

```typescript
// Wrong
const el = document.getElementById("app");
el.innerHTML = "Hello";

// Correct
const el = document.getElementById("app");
if (el) {
  el.innerHTML = "Hello";
}
```

### Fix 2: Use optional chaining

```typescript
// Wrong
const el = document.getElementById("app");
el.innerHTML = "Hello";

// Correct
const el = document.getElementById("app");
el?.innerHTML = "Hello";
```

### Fix 3: Use non-null assertion when you're sure

```typescript
// Use ! only when you're certain the value is not null
const el = document.getElementById("app")!;
el.innerHTML = "Hello";
```

### Fix 4: Use type narrowing in function parameters

```typescript
function process(value: string | null) {
  if (value === null) {
    console.log("No value provided");
    return;
  }
  console.log(value.length); // OK — value is string
}
```

## Examples

```typescript
// Example 1: Document.querySelector returns Element | null
const btn = document.querySelector("button");
btn.addEventListener("click", handler); // TS2531

// Example 2: Object destructuring with optional values
interface Response {
  data?: { name: string };
}
function getName(res: Response) {
  return res.data.name; // TS2531: data is possibly undefined
}
```
