# TS1240 — Unable to resolve signature of property access pattern

TS1240 occurs when TypeScript cannot resolve the type of a property access expression. This happens when the object type does not have the accessed property, or when the type system cannot infer the result of the access pattern.

## Common Causes

```typescript
// Cause 1: Accessing a property on a type that has no index signature
interface NoIndex {
  a: string;
  b: number;
}

declare const obj: NoIndex;
const val = obj["c"]; // TS1240

// Cause 2: Chained property access on an unresolved type
declare const data: any;
const result = data.foo.bar.baz.qux; // TS1240

// Cause 3: Accessing property on union type with incompatible members
type A = { a: string };
type B = { b: number };
declare const ab: A | B;
const val2 = ab.a; // TS1240 — not present on both union members

// Cause 4: Using computed property access with an unresolved expression
const key = Symbol("key");
declare const obj2: { [key]: string };
const val3 = obj2[key]; // TS1240 — symbol indexing may not resolve

// Cause 5: Property access on type parameter without constraint
function process<T>(item: T) {
  const val = item.foo; // TS1240 — T is not constrained
}
```

## Solutions

### Solution 1: Add an index signature or use a type guard

```typescript
// Wrong
interface NoIndex {
  a: string;
  b: number;
}
declare const obj: NoIndex;
const val = obj["c"]; // TS1240

// Correct — add index signature
interface WithIndex {
  a: string;
  b: number;
  [key: string]: string | number;
}
declare const obj2: WithIndex;
const val = obj2["c"]; // OK — returns string | number
```

### Solution 2: Constrain the generic type parameter

```typescript
// Wrong
function process<T>(item: T) {
  const val = item.foo; // TS1240
}

// Correct
function process<T extends { foo: string }>(item: T) {
  const val = item.foo; // OK
}
```

### Solution 3: Use optional chaining or type assertions

```typescript
// Wrong
declare const data: any;
const result = data.foo.bar.baz.qux; // TS1240

// Correct
const result = data?.foo?.bar?.baz?.qux; // OK
```

## Examples

```typescript
interface Config {
  database: {
    host: string;
    port: number;
  };
}

function getHost(config: Config): string {
  return config.database.host; // OK
}

// Using bracket notation
function getPort(config: Config): number {
  return config["database"]["port"]; // OK
}
```
