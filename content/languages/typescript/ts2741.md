# TS2741 — Property 'X' is missing in type 'Y'

TS2741 occurs when an object literal or value is missing required properties that the target type demands. This is a specific variant of TS2322 focused on missing properties in object shapes.

## Common Causes

```typescript
// Cause 1: Missing required property in object literal
interface User {
  name: string;
  age: number;
  email: string;
}
const user: User = { name: "Alice", age: 30 }; // TS2741: Property 'email' is missing

// Cause 2: Missing property in function return
interface Config {
  host: string;
  port: number;
  debug: boolean;
}
function getConfig(): Config {
  return { host: "localhost", port: 3000 }; // TS2741: 'debug' is missing
}

// Cause 3: Missing property when implementing an interface
interface Serializable {
  serialize(): string;
  validate(): boolean;
}
class MyClass implements Serializable {
  serialize() { return ""; } // TS2741: 'validate' is missing
}

// Cause 4: Missing property in array of objects
const users: User[] = [
  { name: "Alice", age: 30 }, // TS2741: 'email' is missing
];
```

## Solutions

### Fix 1: Add all required properties

```typescript
// Wrong
const user: User = { name: "Alice", age: 30 };

// Correct
const user: User = { name: "Alice", age: 30, email: "alice@example.com" };
```

### Fix 2: Make optional properties with ?

```typescript
// If email isn't always needed, mark it optional
interface User {
  name: string;
  age: number;
  email?: string; // Optional
}
const user: User = { name: "Alice", age: 30 }; // OK
```

### Fix 3: Use default values or spread

```typescript
function getConfig(): Config {
  const defaults = { debug: false };
  return { ...defaults, host: "localhost", port: 3000 };
}
```

### Fix 4: Implement all interface members

```typescript
class MyClass implements Serializable {
  serialize() { return ""; }
  validate() { return true; } // Implement missing method
}
```

## Examples

```typescript
// Example 1: React component props
interface ButtonProps {
  onClick: () => void;
  label: string;
  variant: "primary" | "secondary";
}
// Missing 'variant' prop
<Button onClick={handler} label="Click" />; // TS2741

// Example 2: API response type
interface ApiResponse {
  status: number;
  data: unknown;
  timestamp: string;
}
const res: ApiResponse = { status: 200, data: {} }; // TS2741
```
