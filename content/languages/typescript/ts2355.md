# TS2355 — A function whose declared type is neither 'void' nor 'any' must return a value

TS2355 occurs when a function declares a return type that requires a value (like `string`, `number`, `boolean`, etc.) but no `return` statement with a value is present on all code paths.

## Common Causes

```typescript
// Cause 1: Missing return statement
function getValue(): string {
  // no return
} // TS2355

// Cause 2: Return only in some branches
function getStatus(code: number): string {
  if (code === 200) {
    return "OK";
  }
  // TS2355 — no return for other codes
}

// Cause 3: Async function missing return
async function fetchData(): Promise<string> {
  await fetch("https://api.example.com");
  // TS2355 — no return
}

// Cause 4: Function with never reachable end
function process(value: string): number {
  switch (value) {
    case "a":
      return 1;
    case "b":
      return 2;
    // TS2355 — missing default or return
  }
}

// Cause 5: Arrow function without implicit return
const double = (x: number): number => {
  const result = x * 2;
  // TS2355 — missing return
};
```

## Solutions

### Solution 1: Add a return statement

```typescript
// Wrong
function getValue(): string {
  // no return
} // TS2355

// Correct
function getValue(): string {
  return "hello";
}
```

### Solution 2: Handle all code paths

```typescript
// Wrong
function getStatus(code: number): string {
  if (code === 200) {
    return "OK";
  }
} // TS2355

// Correct
function getStatus(code: number): string {
  if (code === 200) {
    return "OK";
  }
  return "Error";
}
```

### Solution 3: Add a default case in switch statements

```typescript
// Wrong
function process(value: string): number {
  switch (value) {
    case "a":
      return 1;
    case "b":
      return 2;
  }
} // TS2355

// Correct
function process(value: string): number {
  switch (value) {
    case "a":
      return 1;
    case "b":
      return 2;
    default:
      return 0;
  }
}
```

### Solution 4: Return in arrow functions

```typescript
// Wrong
const double = (x: number): number => {
  const result = x * 2;
}; // TS2355

// Correct
const double = (x: number): number => {
  return x * 2;
};

// Or use concise body
const double2 = (x: number): number => x * 2;
```

## Examples

```typescript
function divide(a: number, b: number): number {
  if (b === 0) {
    throw new Error("Cannot divide by zero");
  }
  return a / b;
}

// Exhaustive check
function describe(input: "a" | "b"): string {
  switch (input) {
    case "a":
      return "Alpha";
    case "b":
      return "Bravo";
    default:
      const _exhaustive: never = input;
      return _exhaustive;
  }
}
```
