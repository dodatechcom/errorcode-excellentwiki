# TS2430 — Interface 'X' cannot simultaneously extend types 'Y' and 'Z'

TS2430 occurs when an interface extends two or more interfaces that have conflicting member definitions. The inherited members must be compatible across all parent types.

## Common Causes

```typescript
// Cause 1: Conflicting property types
interface A {
  value: string;
}

interface B {
  value: number;
}

interface C extends A, B {} // TS2430 — value is string and number

// Cause 2: Conflicting method signatures
interface Logger {
  log(message: string): void;
}

interface StrictLogger {
  log(message: string, level: number): void;
}

interface AppLogger extends Logger, StrictLogger {} // TS2430

// Cause 3: Conflicting property presence (required vs optional)
interface Required {
  name: string;
}

interface Optional {
  name?: string;
}

interface Mixed extends Required, Optional {} // TS2430

// Cause 4: Conflicting index signatures
interface StringMap {
  [key: string]: string;
}

interface NumberMap {
  [key: string]: number;
}

interface BothMaps extends StringMap, NumberMap {} // TS2430

// Cause 5: Conflicting read-only modifiers
interface Readonly {
  readonly id: number;
}

interface Mutable {
  id: number;
}

interface Mixed2 extends Readonly, Mutable {} // TS2430
```

## Solutions

### Solution 1: Resolve the conflict manually

```typescript
// Wrong
interface A { value: string; }
interface B { value: number; }
interface C extends A, B {} // TS2430

// Correct — redefine the member in C
interface A { value: string; }
interface B { value: number; }
interface C {
  value: string | number;
}
```

### Solution 2: Use a single base interface

```typescript
// Wrong
interface Logger {
  log(message: string): void;
}
interface StrictLogger {
  log(message: string, level: number): void;
}
interface AppLogger extends Logger, StrictLogger {} // TS2430

// Correct
interface AppLogger {
  log(message: string, level?: number): void;
}
```

### Solution 3: Use intersection types with compatible members

```typescript
// Wrong
interface A { value: string; }
interface B { value: number; }

// Correct — use a union where values differ
type Common = {
  value: string | number;
};
```

## Examples

```typescript
// Valid — non-conflicting extensions
interface Printable {
  print(): void;
}

interface Serializable {
  serialize(): string;
}

// No conflict — different member names
interface Document extends Printable, Serializable {
  title: string;
}

// Resolving conflicts
interface A { data: string; }
interface B { data: number; }
interface C extends A {
  // Override to resolve
  data: string | number;
}
```
