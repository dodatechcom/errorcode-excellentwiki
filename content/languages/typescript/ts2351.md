# TS2351 — This expression is not constructable

TS2351 occurs when you try to use `new` on a value whose type doesn't have a construct signature. This happens when calling `new` on a function that isn't a constructor, an object, or a primitive.

## Common Causes

```typescript
// Cause 1: Using 'new' on a regular function
function greet(name: string) {
  return `Hello, ${name}`;
}
const g = new greet("Alice"); // TS2351: not constructable

// Cause 2: Using 'new' on an arrow function
const create = () => ({ x: 1 });
const obj = new create(); // TS2351

// Cause 3: Using 'new' on a module default export that isn't a class
import utils from "./utils";
const u = new utils(); // TS2351

// Cause 4: Using 'new' on an interface
interface Factory {
  create(): Widget;
}
const f = new Factory(); // TS2351: interfaces aren't values
```

## Solutions

### Fix 1: Use a class with a constructor

```typescript
// Wrong — regular function
function Greeter(name: string) {
  this.name = name;
}
new Greeter("Alice"); // TS2351

// Correct — use a class
class Greeter {
  name: string;
  constructor(name: string) {
    this.name = name;
  }
}
new Greeter("Alice"); // OK
```

### Fix 2: Use factory function without 'new'

```typescript
// Wrong
const create = () => ({ x: 1 });
const obj = new create();

// Correct — call without 'new'
const obj = create();
```

### Fix 3: Import the class correctly

```typescript
// Wrong — importing non-class default export
import utils from "./utils";
new utils();

// Correct — import a class
import { MyClass } from "./my-class";
new MyClass();
```

### Fix 4: Define a construct signature for the type

```typescript
interface Constructor {
  new (name: string): { name: string };
}

function createInstance(Ctor: Constructor, name: string) {
  return new Ctor(name); // OK
}
```

## Examples

```typescript
// Example 1: Using 'new' on a function stored in a variable
const initialize = function () {};
const app = new initialize(); // TS2351

// Example 2: Using 'new' on a namespace
namespace Utils {
  export function helper() {}
}
new Utils(); // TS2351
```
