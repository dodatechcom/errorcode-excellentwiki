# TS7045 — Variable declared but never used

TS7045 occurs when a variable, function, or type is declared but never referenced anywhere in the code, with the `noUnusedLocals` strict variant. This is a stricter version of TS6133 that also catches unused types and parameters in certain contexts.

## Common Causes

```typescript
// Cause 1: Variable declared but never used
function render() {
  const unusedVar = "hello"; // TS7045
  return "rendered";
}

// Cause 2: Function parameter not used
function process(id: number, name: string): number {
  // TS7045 — 'name' is declared but never used
  return id * 2;
}

// Cause 3: Type alias declared but never used
type UserStatus = "active" | "inactive"; // TS7045

// Cause 4: Import used only as type but imported as value
import { SomeType } from "./module"; // TS7045 — only used as type
```

## Solutions

### Fix 1: Remove the unused declaration

```typescript
// Wrong
function render() {
  const unusedVar = "hello";
  return "rendered";
}

// Correct
function render() {
  return "rendered";
}
```

### Fix 2: Prefix with underscore for intentional unused params

```typescript
// Wrong
function process(id: number, name: string): number {
  return id * 2;
}

// Correct
function process(id: number, _name: string): number {
  return id * 2;
}
```

### Fix 3: Use type import syntax

```typescript
// Wrong
import { SomeType } from "./module";

// Correct — only import the type
import type { SomeType } from "./module";
```

### Fix 4: Use the variable or remove it

```typescript
// Use it:
function render() {
  const unusedVar = "hello";
  console.log(unusedVar); // now used
  return "rendered";
}

// Or remove it entirely
function render() {
  return "rendered";
}
```

## Examples

```typescript
// Example 1: Destructured unused parameter
function handleEvent({ target }: { target: EventTarget }) {
  // TS7045 if 'target' is used but another destructured var is not
}

// Example 2: Unused type parameter
function factory<T>(value: number): number {
  // TS7045 — T is declared but never used
  return value * 2;
}

// Fix: remove T if not needed
function factory(value: number): number {
  return value * 2;
}
```
