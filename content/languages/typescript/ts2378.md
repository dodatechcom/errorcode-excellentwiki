# TS2378 — Getter and setter types must match

TS2378 occurs when a getter and setter for the same property have incompatible types. The setter's parameter type must be assignable to the getter's return type and vice versa.

## Common Causes

```typescript
// Cause 1: Getter returns string, setter accepts boolean
class Config {
  private _enabled: string = "yes";

  get enabled(): string {
    return this._enabled;
  }

  set enabled(value: boolean) { // TS2378
    this._enabled = value ? "yes" : "no";
  }
}

// Cause 2: Incompatible union types
class State {
  private _status: string = "idle";

  get status(): "idle" | "running" {
    return this._status as "idle" | "running";
  }

  set status(value: "pending" | "done") { // TS2378
    this._status = value;
  }
}

// Cause 3: Getter returns optional, setter accepts non-optional (different direction)
class Data {
  private _value: number | undefined;

  get value(): number {
    return this._value!; // TS2378 — getter says always number
  }

  set value(v: number | undefined) { // TS2378 — setter allows undefined
    this._value = v;
  }
}

// Cause 4: Generic type mismatch
class Box<T> {
  private _value!: T;

  get value(): T {
    return this._value;
  }

  set value(v: string) { // TS2378
    this._value = v as T;
  }
}
```

## Solutions

### Solution 1: Use the same type for both getter and setter

```typescript
// Wrong
class Config {
  private _enabled: string = "yes";
  get enabled(): string {
    return this._enabled;
  }
  set enabled(value: boolean) { // TS2378
    this._enabled = value ? "yes" : "no";
  }
}

// Correct
class Config {
  private _enabled: boolean = true;
  get enabled(): boolean {
    return this._enabled;
  }
  set enabled(value: boolean) {
    this._enabled = value;
  }
}
```

### Solution 2: Use compatible union types

```typescript
// Wrong
class State {
  private _status: string = "idle";
  get status(): "idle" | "running" {
    return this._status as "idle" | "running";
  }
  set status(value: "pending" | "done") { // TS2378
    this._status = value;
  }
}

// Correct
class State {
  private _status: "idle" | "running" = "idle";
  get status(): "idle" | "running" {
    return this._status;
  }
  set status(value: "idle" | "running") {
    this._status = value;
  }
}
```

### Solution 3: Ensure both directions are compatible

```typescript
// Wrong
class Data {
  private _value: number | undefined;
  get value(): number {
    return this._value!;
  }
  set value(v: number | undefined) { // TS2378
    this._value = v;
  }
}

// Correct
class Data {
  private _value: number | undefined;
  get value(): number | undefined {
    return this._value;
  }
  set value(v: number | undefined) {
    this._value = v;
  }
}
```

## Examples

```typescript
class Temperature {
  private _celsius: number = 0;

  get fahrenheit(): number {
    return this._celsius * 9 / 5 + 32;
  }

  set fahrenheit(value: number) {
    this._celsius = (value - 32) * 5 / 9;
  }

  get celsius(): number {
    return this._celsius;
  }

  set celsius(value: number) {
    this._celsius = value;
  }
}
```
