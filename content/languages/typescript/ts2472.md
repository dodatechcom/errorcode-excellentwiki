# TS2472 — Spread types may only be created from object types

TS2472 occurs when you use the spread operator (`...`) on a value that is not an object type. The spread operator only works with objects (including arrays, which are a subtype of object).

## Common Causes

```typescript
// Cause 1: Spreading a string
const str = "hello";
const obj = { ...str }; // TS2472

// Cause 2: Spreading a number
const num = 42;
const obj = { ...num }; // TS2472

// Cause 3: Spreading a union type that includes non-objects
const value: string | object = "hello";
const obj = { ...value }; // TS2472 — string is not an object

// Cause 4: Spreading a null or undefined
const data: object | null = null;
const obj = { ...data }; // TS2472

// Cause 5: Spreading a function return type that might not be an object
function getData(): Record<string, unknown> | null {
  return null;
}
const merged = { ...getData() }; // TS2472
```

## Solutions

### Solution 1: Ensure the value is an object type

```typescript
// Wrong
const str = "hello";
const obj = { ...str }; // TS2472

// Correct
const obj = { ...{ key: "value" } };
```

### Solution 2: Add a type guard or null check

```typescript
// Wrong
const value: string | object = "hello";
const obj = { ...value }; // TS2472

// Correct
if (typeof value === "object" && value !== null) {
  const obj = { ...value };
}
```

### Solution 3: Use Object.assign instead

```typescript
// Wrong
const data: Record<string, unknown> | null = null;
const merged = { ...data }; // TS2472

// Correct
const merged = Object.assign({}, data ?? {});
```

### Solution 4: Provide a default empty object

```typescript
// Wrong
function getData(): Record<string, unknown> | null {
  return null;
}
const merged = { ...getData() }; // TS2472

// Correct
const merged = { ...(getData() ?? {}) };
```

## Examples

```typescript
// Valid spreads
const defaults = { theme: "dark", lang: "en" };
const overrides = { theme: "light" };
const config = { ...defaults, ...overrides }; // { theme: "light", lang: "en" }

// Spreading arrays (arrays are objects)
const arr1 = [1, 2, 3];
const arr2 = [...arr1, 4, 5]; // [1, 2, 3, 4, 5]

// Spreading function result (when guaranteed to be object)
function getSettings(): Record<string, string> {
  return { mode: "edit" };
}
const settings = { ...getSettings(), mode: "view" };
```
