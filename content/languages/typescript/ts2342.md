# TS2342 — Return type annotation circularity

TS2342 occurs when a function's return type annotation creates a circular reference. TypeScript cannot resolve the type because it depends on itself directly or indirectly.

## Common Causes

```typescript
// Cause 1: Function returning its own type annotation
function create(): ReturnType<typeof create> { // TS2342
  return create;
}

// Cause 2: Indirect circular return type
function a(): ReturnType<typeof b> { return b(); } // TS2342
function b(): ReturnType<typeof a> { return a(); }

// Cause 3: Return type referencing a type that cycles back
type Node = {
  children: Node[];
  parent: ReturnType<typeof getParent>; // TS2342
};

function getParent(node: Node): Node | null {
  return null;
}

// Cause 4: Recursive function with circular return type annotation
function factorial(n: number): typeof factorial { // TS2342
  return factorial;
}

// Cause 5: Arrow function with inferred circular type
const fn = (): ReturnType<typeof fn> => fn(); // TS2342
```

## Solutions

### Solution 1: Use a concrete return type instead of ReturnType

```typescript
// Wrong
function create(): ReturnType<typeof create> { // TS2342
  return create;
}

// Correct
type Factory = () => Factory;
function create(): Factory {
  return create;
}
```

### Solution 2: Break the cycle with an explicit type

```typescript
// Wrong
function a(): ReturnType<typeof b> { return b(); } // TS2342
function b(): ReturnType<typeof a> { return a(); }

// Correct
type Handler = () => void;
function a(): Handler { return b; }
function b(): Handler { return a; }
```

### Solution 3: Use a named type for the return

```typescript
// Wrong
const fn = (): ReturnType<typeof fn> => fn(); // TS2342

// Correct
type Self = () => Self;
const fn: Self = () => fn;
```

### Solution 4: Remove the circular annotation entirely

```typescript
// Wrong
function factorial(n: number): typeof factorial { // TS2342
  return factorial;
}

// Correct
function factorial(n: number): number {
  return n <= 1 ? 1 : n * factorial(n - 1);
}
```

## Examples

```typescript
// Valid recursive types with no circularity
interface TreeNode {
  value: number;
  children: TreeNode[];
}

function createTree(): TreeNode {
  return {
    value: 1,
    children: [
      { value: 2, children: [] },
      { value: 3, children: [] },
    ],
  };
}

// Valid self-referential function type
type JsonParser = (input: string) => JsonParser;
const parser: JsonParser = (input) => parser;
```
