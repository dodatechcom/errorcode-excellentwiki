# TS1238 — Unable to resolve signature of class declaration

TS1238 occurs when TypeScript cannot resolve the signature of a class declaration. This typically happens when a class extends something that does not resolve to a valid constructor type, or when the constructor signature is malformed.

## Common Causes

```typescript
// Cause 1: Extending a non-constructor expression
const notAClass = 42;
class Child extends notAClass {} // TS1238

// Cause 2: Extending a function that does not return an object type
function factory(): number {
  return 42;
}
class Child extends factory {} // TS1238

// Cause 3: Extending an expression that resolves to undefined
const maybeClass = undefined;
class Child extends maybeClass {} // TS1238

// Cause 4: Extending a namespace used as a type
namespace NS {
  export interface Thing {
    value: string;
  }
}
class Child extends NS {} // TS1238

// Cause 5: Extending a null value
const base = null;
class Child extends base {} // TS1238
```

## Solutions

### Solution 1: Extend a valid class

```typescript
// Wrong
const notAClass = 42;
class Child extends notAClass {} // TS1238

// Correct
class Base {
  greet() {
    return "hello";
  }
}
class Child extends Base {}
```

### Solution 2: Ensure the expression resolves to a constructor

```typescript
// Wrong
function factory(): number {
  return 42;
}
class Child extends factory {} // TS1238

// Correct
class Base {
  value: number;
  constructor(v: number) {
    this.value = v;
  }
}

function getBase(): typeof Base {
  return Base;
}

class Child extends getBase() {} // Valid
```

### Solution 3: Use a valid class or constructor function

```typescript
// Wrong
const base = undefined;
class Child extends base {} // TS1238

// Correct
class Base {}
class Child extends Base {}
```

## Examples

```typescript
class Animal {
  name: string;
  constructor(name: string) {
    this.name = name;
  }
}

class Dog extends Animal {
  breed: string;
  constructor(name: string, breed: string) {
    super(name);
    this.breed = breed;
  }
}

const dog = new Dog("Rex", "Labrador");
console.log(dog.name); // "Rex"
```
