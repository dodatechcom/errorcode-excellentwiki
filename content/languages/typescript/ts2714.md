# TS2714 — Expression expected with 'typeof'

TS2714 occurs when `typeof` is used in a context where a type is expected but the syntax is invalid. This often happens when `typeof` is used incorrectly in type annotations or when an expression is expected after `typeof` in a value position.

## Common Causes

```typescript
// Cause 1: typeof used incorrectly in type annotation
const x: typeof = 42; // TS2714 — typeof needs an expression

// Cause 2: typeof in wrong position
type X = typeof; // TS2714 — no expression

// Cause 3: Missing expression after typeof
function process(value: typeof ) {} // TS2714

// Cause 4: typeof used as a value expression incorrectly
const type = typeof; // TS2714 — typeof needs an operand

// Cause 5: typeof in template literal without expression
type Y = `${typeof}`; // TS2714
```

## Solutions

### Solution 1: Provide an expression after typeof

```typescript
// Wrong
const x: typeof = 42; // TS2714

// Correct
const myVar = "hello";
const x: typeof myVar = "world"; // OK
```

### Solution 2: Use typeof with a variable in type positions

```typescript
// Wrong
type X = typeof; // TS2714

// Correct
const config = { debug: true };
type X = typeof config; // { debug: boolean }
```

### Solution 3: Use typeof in function parameter types

```typescript
// Wrong
function process(value: typeof) {} // TS2714

// Correct
const defaults = { theme: "dark" };
function process(value: typeof defaults) {
  console.log(value.theme);
}
```

## Examples

```typescript
// typeof in type alias
const defaultSettings = {
  host: "localhost",
  port: 3000,
  debug: false,
};
type Settings = typeof defaultSettings;

// typeof in mapped types
const person = { name: "Alice", age: 30 };
type PersonKeys = keyof typeof person; // "name" | "age"

// typeof in function return type
function getConfig() {
  return { debug: true, verbose: false };
}
type Config = ReturnType<typeof getConfig>;

// typeof with class instances
class UserService {
  findById(id: string) { return { id, name: "User" }; }
}
type User = ReturnType<InstanceType<typeof UserService>["findById"]>;
```
