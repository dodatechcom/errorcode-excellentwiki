# TS1206 — Decorators are not valid here

TypeScript decorators have strict placement rules. They can only appear on class declarations, class members (methods, properties, accessors), or constructor parameters. Using a decorator in an unsupported position triggers TS1206.

## Common Causes

```typescript
// Cause 1: Decorator on a function declaration
function log(target: any, key: string, descriptor: PropertyDescriptor) {
  return descriptor;
}

@log // TS1206 — cannot decorate a standalone function declaration
function doSomething() {}

// Cause 2: Decorator on an interface
@interface Validatable {
  validate(): void;
}

@Validatable // TS1206
class Form {}

// Cause 3: Decorator on a variable
@sealed
const myObj = {}; // TS1206

// Cause 4: Decorator on a type alias
@deprecated
type OldType = string; // TS1206

// Cause 5: Decorator used in a file without experimentalDecorators enabled
class Example {
  @log // TS1206 (if experimentalDecorators is not set)
  method() {}
}
```

## Solutions

### Solution 1: Move the decorator to a class or class member

```typescript
// Wrong
function log(target: any, key: string, descriptor: PropertyDescriptor) {
  return descriptor;
}

@log
function doSomething() {} // TS1206

// Correct — decorate a class method instead
class Service {
  @log
  doSomething() {}
}
```

### Solution 2: Enable experimentalDecorators in tsconfig.json

```json
{
  "compilerOptions": {
    "experimentalDecorators": true,
    "emitDecoratorMetadata": true
  }
}
```

### Solution 3: Decorate class properties, methods, accessors, or the class itself

```typescript
// Wrong — decorator on a standalone property declaration outside a class
@log
const x = 10; // TS1206

// Correct — decorator on a class property
class MyClass {
  @log
  x = 10; // Valid
}
```

## Examples

```typescript
function sealed(constructor: Function) {
  Object.seal(constructor);
  Object.seal(constructor.prototype);
}

function readonly(target: any, propertyKey: string) {
  Object.defineProperty(target, propertyKey, { writable: false });
}

@sealed
class Greeter {
  greeting: string;

  constructor(message: string) {
    this.greeting = message;
  }

  @readonly
  greet() {
    return `Hello, ${this.greeting}`;
  }
}
```
