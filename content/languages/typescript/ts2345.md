# TS2345 — Argument of type 'X' is not assignable

TS2345 occurs when you pass an argument to a function that doesn't match the expected parameter type. This is one of the most common TypeScript errors and covers a wide range of type incompatibilities.

## Common Causes

```typescript
// Cause 1: Wrong primitive type
function greet(name: string) {}
greet(42); // TS2345: Argument of type 'number' is not assignable to parameter of type 'string'

// Cause 2: Missing required properties in object argument
function createUser(user: { name: string; age: number }) {}
createUser({ name: "Alice" }); // TS2345: Property 'age' is missing

// Cause 3: Array type mismatch
function processItems(items: number[]) {}
processItems(["a", "b"]); // TS2345: Type 'string[]' not assignable to 'number[]'

// Cause 4: Nullable argument to non-nullable parameter
function handle(input: string) {}
handle(null); // TS2345 if strictNullChecks is on
```

## Solutions

### Fix 1: Match the expected argument type

```typescript
// Wrong
function greet(name: string) {}
greet(42);

// Correct
greet("Alice");
```

### Fix 2: Provide all required properties

```typescript
// Wrong
function createUser(user: { name: string; age: number }) {}
createUser({ name: "Alice" });

// Correct
createUser({ name: "Alice", age: 30 });
```

### Fix 3: Cast or convert arguments to the expected type

```typescript
function processItems(items: number[]) {}
processItems(["1", "2"].map(Number)); // Convert strings to numbers
```

### Fix 4: Use union types to accept multiple types

```typescript
function handle(input: string | null) {
  if (input === null) return;
  // input is string here
}
handle(null); // OK
```

## Examples

```typescript
// Example 1: Incompatible object literal
interface Config {
  host: string;
  port: number;
}
function startServer(config: Config) {}
startServer({ host: "localhost" }); // TS2345

// Example 2: Array vs single value
function log(items: string[]) {}
log("hello"); // TS2345: 'string' is not assignable to 'string[]'
```
