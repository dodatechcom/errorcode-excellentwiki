# TS7027 — Unreachable code detected

TS7027 occurs when TypeScript detects code that can never be executed. This typically happens after `return`, `throw`, `break`, or `continue` statements, or when a condition is always true or false.

## Common Causes

```typescript
// Cause 1: Code after return statement
function greet(name: string) {
  return `Hello, ${name}`;
  console.log("This will never run"); // TS7027
}

// Cause 2: Code after throw
function validate(input: string) {
  throw new Error("Invalid");
  console.log("After throw"); // TS7027
}

// Cause 3: Code after break/continue in loop
function findValue(arr: number[]) {
  for (const item of arr) {
    if (item === 5) {
      break;
      console.log("Found!"); // TS7027
    }
  }
}

// Cause 4: Code after always-throw or always-return condition
function process(x: number) {
  if (x > 0) {
    return x;
  } else {
    throw new Error("negative");
  }
  console.log("done"); // TS7027 — both branches exit
}
```

## Solutions

### Fix 1: Remove unreachable code

```typescript
// Wrong
function greet(name: string) {
  return `Hello, ${name}`;
  console.log("This will never run");
}

// Correct
function greet(name: string) {
  return `Hello, ${name}`;
}
```

### Fix 2: Move code before the return/throw

```typescript
// Wrong
function validate(input: string) {
  throw new Error("Invalid");
  console.log("Logging");
}

// Correct
function validate(input: string) {
  console.log("Logging");
  throw new Error("Invalid");
}
```

### Fix 3: Fix infinite loops or always-false conditions

```typescript
// Wrong
while (true) {
  doSomething();
}
console.log("After loop"); // TS7027 — while(true) never ends

// Correct
let running = true;
while (running) {
  doSomething();
  running = shouldContinue();
}
console.log("After loop");
```

## Examples

```typescript
// Example 1: Debug logging after return
function calculate(a: number, b: number) {
  const result = a + b;
  console.log(result); // This should be before return
  return result;
}

// Example 2: Wrong branch placement
function check(value: number) {
  if (value < 0) return "negative";
  if (value > 0) return "positive";
  return "zero";
  console.log("checked"); // TS7027
}
```
