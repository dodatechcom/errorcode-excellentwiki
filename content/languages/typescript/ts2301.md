# TS2301 — Duplicate function implementation

TS2301 occurs when the same function is implemented more than once in the same scope. This differs from function overloads, which require different parameter signatures. Duplicate implementations mean the exact same signature appears twice.

## Common Causes

```typescript
// Cause 1: Defining the same function twice
function greet(name: string) {
  return `Hello, ${name}`;
}
function greet(name: string) { // TS2301: Duplicate function implementation
  return `Hi, ${name}`;
}

// Cause 2: Arrow function reassigned in same scope
const add = (a: number, b: number) => a + b;
const add = (a: number, b: number) => a + b; // TS2301

// Cause 3: Method defined twice in a class
class Service {
  process() { return 1; }
  process() { return 2; } // TS2301
}

// Cause 4: Duplicate function in module augmentation
declare module "foo" {
  function bar(): void;
  function bar(): string; // TS2301 — same signature
}
```

## Solutions

### Fix 1: Remove the duplicate implementation

```typescript
// Wrong — two implementations
function greet(name: string) {
  return `Hello, ${name}`;
}
function greet(name: string) {
  return `Hi, ${name}`;
}

// Correct — keep only one
function greet(name: string) {
  return `Hello, ${name}`;
}
```

### Fix 2: Use function overloads if you need multiple signatures

```typescript
// Wrong — same signature twice
function process(input: string): string;
function process(input: string): string; // TS2301

// Correct — different signatures
function process(input: string): string;
function process(input: number): string;
function process(input: string | number): string {
  return String(input);
}
```

### Fix 3: Rename duplicate arrow functions

```typescript
// Wrong
const add = (a: number, b: number) => a + b;
const add = (a: number, b: number) => a + b;

// Correct — use a different name
const add = (a: number, b: number) => a + b;
const addVerbose = (a: number, b: number) => a + b;
```

### Fix 4: Merge method implementations

```typescript
// Wrong
class Service {
  process() { return 1; }
  process() { return 2; }
}

// Correct
class Service {
  process(mode?: string) {
    return mode === "fast" ? 2 : 1;
  }
}
```

## Examples

```typescript
// Example 1: Accidental duplicate in module
// utils.ts
export function formatDate(d: Date) { return d.toISOString(); }
export function formatDate(d: Date) { return d.toLocaleDateString(); } // TS2301

// Example 2: Namespace and function conflict
function build() {}
namespace build {} // May cause TS2301 in some configurations
```
