# TS2340 — Only void and generic functions are callable

TS2340 occurs when you try to call a function-like expression (e.g., `new Foo()`) that evaluates to a type with no construct signature, but does have a call signature that is only valid for non-void/void returns. This typically appears with class-like values that aren't actually constructable.

## Common Causes

```typescript
// Cause 1: Calling a plain object as a constructor
const obj = {};
new obj(); // TS2340

// Cause 2: Calling an interface without new signature
interface MyType {
  (): void;
}
new MyType(); // TS2340

// Cause 3: Calling a non-generic, non-void callable with new
interface Callable {
  (x: number): number;
}
new Callable(); // TS2340 — no construct signature
```

## Solutions

### Fix 1: Only use `new` with constructors

```typescript
// Wrong
const obj = {};
new obj();

// Correct
class MyClass {}
const instance = new MyClass();
```

### Fix 2: Add a construct signature to the type

```typescript
interface Constructable {
  new (): MyType;
  (): void;
}

// Now it can be called with new
function create(ctor: Constructable) {
  return new ctor();
}
```

### Fix 3: Call without `new` if the function returns a value

```typescript
// Wrong
class MyClass {
  constructor() { return {}; }
}
const instance = new MyClass(); // Works but not what you expect

// If the function is meant to be called without new:
function createInstance() { return {}; }
const instance = createInstance();
```

## Examples

```typescript
// Example 1: Using a mixin that returns a constructor
function Mixin<T extends new (...args: any[]) => any>(Base: T) {
  return class extends Base {};
}

const Mixed = Mixin(SomeClass);
new Mixed(); // OK — Mixin returns a constructable type
```
