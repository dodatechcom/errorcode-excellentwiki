# TS2314 — Generic type 'X' requires N type argument(s)

TS2314 occurs when you use a generic type without providing the required number of type arguments. Every type parameter defined on the generic must be supplied.

## Common Causes

```typescript
// Cause 1: Missing type argument on generic type
interface Box<T> {
  value: T;
}

const box: Box = { value: 42 }; // TS2314 — requires 1 type argument

// Cause 2: Missing type arguments on generic function type
type Mapper<T, U> = (input: T) => U;

const fn: Mapper = (x) => x; // TS2314 — requires 2 type arguments

// Cause 3: Missing type argument on generic class
class Container<T> {
  value!: T;
}

const c: Container = { value: "hello" }; // TS2314

// Cause 4: Missing type argument on generic interface in extends clause
interface Base<T> {
  value: T;
}

// interface Child extends Base {} // TS2314

// Cause 5: Generic used in type annotation without arguments
type Pair<A, B> = [A, B];
const p: Pair = [1, "a"]; // TS2314
```

## Solutions

### Solution 1: Provide all required type arguments

```typescript
// Wrong
interface Box<T> {
  value: T;
}
const box: Box = { value: 42 }; // TS2314

// Correct
const box: Box<number> = { value: 42 };
```

### Solution 2: Provide type arguments to generic function types

```typescript
// Wrong
type Mapper<T, U> = (input: T) => U;
const fn: Mapper = (x) => x; // TS2314

// Correct
const fn: Mapper<number, string> = (x) => x.toString();
```

### Solution 3: Use type inference where possible

```typescript
// Wrong
function identity<T>(x: T): T {
  return x;
}

const fn: typeof identity = (x) => x; // This works — TS can infer
// But if you must annotate:
const fn2: <T>(x: T) => T = identity; // OK
```

### Solution 4: Default type parameters

```typescript
// Wrong
interface Response<T> {
  data: T;
}
const r: Response = { data: "ok" }; // TS2314

// Correct — add default
interface Response<T = unknown> {
  data: T;
}
const r: Response = { data: "ok" }; // OK — defaults to unknown
```

## Examples

```typescript
interface Result<T, E = Error> {
  value?: T;
  error?: E;
}

// Providing both type arguments
const success: Result<string, never> = { value: "done" };
const failure: Result<never, TypeError> = { error: new TypeError("bad") };

// Using default for E
const simple: Result<number> = { value: 42 };
```
