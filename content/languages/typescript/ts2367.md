# TS2367 — This comparison appears to be unintentional

TS2367 occurs when a comparison is always `true` or always `false` because the two sides of the comparison have types with no overlap. TypeScript flags these as likely bugs — you're comparing values that can never be equal (or always are).

## Common Causes

```typescript
// Cause 1: Comparing a type with a value it can never hold
let status: "active" | "inactive";
if (status === "deleted") { // TS2367: This comparison appears to be unintentional
  // "deleted" is not in the union
}

// Cause 2: Comparing a string literal union with an invalid value
type Color = "red" | "green" | "blue";
const isRed = color === "yellow"; // TS2367

// Cause 3: Comparing a number enum with a non-existent member
enum Direction { Up, Down }
const isLeft = Direction.Up === 2; // TS2367: Comparison always false

// Cause 4: Comparing 'never' type with anything
function exhaustive(x: never) {
  if (x === "something") {} // TS2367
}
```

## Solutions

### Fix 1: Use valid values from the union

```typescript
// Wrong
let status: "active" | "inactive";
if (status === "deleted") {} // TS2367

// Correct
if (status === "active") {} // Valid comparison
```

### Fix 2: Widen the type to accept the comparison value

```typescript
// Wrong
let status: "active" | "inactive";

// Correct — if "deleted" is a valid state, add it to the union
let status: "active" | "inactive" | "deleted";
if (status === "deleted") {} // OK
```

### Fix 3: Remove impossible branches

```typescript
type Fruit = "apple" | "banana";

function describe(fruit: Fruit) {
  if (fruit === "orange") {
    // This branch is unreachable — TS2367
    return "This never happens";
  }
  return fruit; // Just handle valid cases
}
```

### Fix 4: Use a type guard function for runtime checks

```typescript
function isString(value: unknown): value is string {
  return typeof value === "string";
}

let x: number | string;
if (isString(x)) {
  // x is string here
}
```

## Examples

```typescript
// Example 1: Exhaustive switch without proper typing
type Shape = "circle" | "square";
function area(shape: Shape) {
  switch (shape) {
    case "circle": return Math.PI;
    case "square": return 1;
    case "triangle": return 0; // TS2367
  }
}

// Example 2: Comparing different enum types
enum A { X, Y }
enum B { X, Y }
const eq = A.X === B.X; // TS2367: A.X and B.X are different types
```
