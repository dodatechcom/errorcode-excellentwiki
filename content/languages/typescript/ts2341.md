# TS2341 — Property getter/setter type mismatch

TS2341 occurs when the return type of a getter and the parameter type of a setter for the same property do not match. The getter and setter must have compatible types.

## Common Causes

```typescript
// Cause 1: Getter returns string, setter accepts number
class User {
  private _name: string = "";

  get name(): string {
    return this._name;
  }

  set name(value: number) { // TS2341 — setter type doesn't match getter
    this._name = String(value);
  }
}

// Cause 2: Getter returns number, setter accepts string
class Counter {
  private _count: number = 0;

  get count(): number {
    return this._count;
  }

  set count(value: string) { // TS2341
    this._count = parseInt(value);
  }
}

// Cause 3: Getter returns optional, setter requires non-optional
class Config {
  private _host: string = "";

  get host(): string | undefined {
    return this._host || undefined;
  }

  set host(value: string) { // TS2341 — types are different
    this._host = value;
  }
}

// Cause 4: Incompatible generic types between getter and setter
class Box<T> {
  private _value!: T;

  get value(): T {
    return this._value;
  }

  set value(v: string) { // TS2341
    this._value = v as T;
  }
}
```

## Solutions

### Solution 1: Make setter type match getter type

```typescript
// Wrong
class User {
  private _name: string = "";
  get name(): string {
    return this._name;
  }
  set name(value: number) { // TS2341
    this._name = String(value);
  }
}

// Correct
class User {
  private _name: string = "";
  get name(): string {
    return this._name;
  }
  set name(value: string) {
    this._name = value;
  }
}
```

### Solution 2: Accept the same type with a narrower setter

```typescript
// Wrong
class Counter {
  private _count: number = 0;
  get count(): number {
    return this._count;
  }
  set count(value: string) { // TS2341
    this._count = parseInt(value);
  }
}

// Correct
class Counter {
  private _count: number = 0;
  get count(): number {
    return this._count;
  }
  set count(value: number) {
    this._count = value;
  }
}
```

### Solution 3: Use compatible types in both directions

```typescript
// Wrong
class Config {
  private _host: string = "";
  get host(): string | undefined {
    return this._host || undefined;
  }
  set host(value: string) { // TS2341 — types differ
    this._host = value;
  }
}

// Correct
class Config {
  private _host: string = "";
  get host(): string {
    return this._host;
  }
  set host(value: string) {
    this._host = value;
  }
}
```

## Examples

```typescript
class Temperature {
  private _celsius: number = 0;

  get fahrenheit(): number {
    return this._celsius * 9 / 5 + 32;
  }

  set fahrenheit(value: number) {
    this._celsius = (value - 32) * 5 / 9;
  }

  get celsius(): number {
    return this._celsius;
  }

  set celsius(value: number) {
    this._celsius = value;
  }
}

const temp = new Temperature();
temp.celsius = 100;
console.log(temp.fahrenheit); // 212
temp.fahrenheit = 32;
console.log(temp.celsius); // 0
```
