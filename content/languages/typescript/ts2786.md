# TS2786 — 'X' cannot be used as a JSX component

TS2786 occurs when you use a value as a JSX component, but its type doesn't match the expected JSX component interface. This commonly happens with React components that have incorrect prop types, missing type definitions, or when using third-party component libraries.

## Common Causes

```typescript
// Cause 1: Function component with wrong return type
function MyComponent() {
  return null; // Returns null, which is valid JSX
}
// But if the function signature is wrong:
const MyComponent = () => 42; // TS2786: number is not a valid JSX element

// Cause 2: Using a non-component as a JSX element
const notAComponent = "hello";
return <notAComponent />; // TS2786

// Cause 3: Missing React types
// If @types/react is not installed
const App = () => <div>Hello</div>; // TS2786 if React types missing

// Cause 4: Class component not extending React.Component
class MyComponent {
  render() {
    return <div>Hello</div>; // TS2786: not a valid React component
  }
}
```

## Solutions

### Fix 1: Ensure the component returns a valid JSX element

```typescript
// Wrong
const MyComponent = () => 42;

// Correct — return JSX, null, or a valid element
const MyComponent = () => <div>Hello</div>;
const MyComponent = () => null; // Valid in React
```

### Fix 2: Install React type definitions

```bash
npm install --save-dev @types/react
```

### Fix 3: Extend React.Component for class components

```typescript
// Wrong
class MyComponent {
  render() {
    return <div>Hello</div>;
  }
}

// Correct
import React from "react";
class MyComponent extends React.Component {
  render() {
    return <div>Hello</div>;
  }
}
```

### Fix 4: Use the correct function component signature

```typescript
// Wrong — arrow function without proper typing
const MyComponent = (props) => <div>{props.name}</div>;

// Correct
interface Props {
  name: string;
}
const MyComponent: React.FC<Props> = ({ name }) => <div>{name}</div>;
```

## Examples

```typescript
// Example 1: Forwarded ref component
const MyComponent = React.forwardRef<HTMLDivElement>((props, ref) => {
  return <div ref={ref}>{props.children}</div>;
});
// If forwardRef types are wrong, TS2786 occurs

// Example 2: Higher-order component
function withAuth(Component: React.ComponentType) {
  return (props: any) => {
    if (!isLoggedIn()) return null;
    return <Component {...props} />; // TS2786 if ComponentType is wrong
  };
}
```
