# TS2684 — The 'this' context of type 'X' is not assignable

TS2684 occurs when the actual `this` context in a function doesn't match the declared `this` parameter type. This is stricter than TS2683 — TypeScript knows what `this` is, but it doesn't match what the function expects.

## Common Causes

```typescript
// Cause 1: 'this' parameter type doesn't match the object
interface Logger {
  log: (msg: string) => void;
  prefix: string;
}
function doLog(this: Logger, msg: string) {
  console.log(this.prefix + msg);
}
const obj = { prefix: "[APP]", log: console.log };
doLog.call(obj, "hello"); // TS2684: 'this' context mismatch

// Cause 2: Method called on wrong object type
class Animal {
  speak() {}
}
class Robot {
  code() {}
}
const robot = new Robot();
const speak = Animal.prototype.speak;
speak.call(robot); // TS2684: Robot is not Animal

// Cause 3: Arrow function captures wrong 'this'
class Outer {
  value = 10;
  method() {
    const inner = {
      value: 20,
      getThis: () => this.value // Captures Outer's 'this', not inner's
    };
  }
}

// Cause 4: Calling method with incompatible receiver
interface HasId {
  id: number;
}
function getId(this: HasId) {
  return this.id;
}
getId.call({ name: "Alice" }); // TS2684: missing 'id'
```

## Solutions

### Fix 1: Ensure the object matches the 'this' parameter type

```typescript
interface Logger {
  log: (msg: string) => void;
  prefix: string;
}

// Wrong — obj doesn't fully match Logger
const obj = { prefix: "[APP]", log: console.log };

// Correct — ensure full type match
const obj: Logger = { prefix: "[APP]", log: console.log };
doLog.call(obj, "hello");
```

### Fix 2: Use .bind() with the correct type

```typescript
class Animal {
  speak() { return "speaking"; }
}
const animal = new Animal();
const speak = animal.speak.bind(animal);
speak(); // OK
```

### Fix 3: Use arrow functions to capture the right 'this'

```typescript
class Outer {
  value = 10;
  method() {
    // Arrow function captures Outer's this
    const getOuterValue = () => this.value;
    return getOuterValue();
  }
}
```

### Fix 4: Match 'this' parameter to actual context

```typescript
interface HasId {
  id: number;
}

// Wrong — calling with object that doesn't have 'id'
getId.call({ name: "Alice" });

// Correct — pass object with required shape
getId.call({ id: 1, name: "Alice" });
```

// Example 1: React class component

  }
}

// Example 2: forEach callback with 'this' parameter

arr.forEach(function (item) {
  console.log(this); // 'this' must match declared type
