# TS1225 — Cannot use namespace 'X' as a type

TS1225 occurs when you try to use a namespace name directly as a type annotation. Namespaces contain members — you must reference a specific exported type or interface within the namespace, not the namespace itself.

## Common Causes

```typescript
// Cause 1: Using namespace as a type directly
namespace MyTypes {
  export interface User {
    name: string;
    age: number;
  }
}

function greet(user: MyTypes): void { // TS1225
  console.log(user.name);
}

// Cause 2: Importing a namespace and using it as a type
import * as Models from "./models";
function process(item: Models): void {} // TS1225

// Cause 3: Namespace re-exported as a namespace used as a type
namespace Config {
  export interface Settings {
    theme: string;
  }
}

const settings: Config = { theme: "dark" }; // TS1225

// Cause 4: Misusing namespace in a generic
namespace Helpers {
  export interface Options {
    verbose: boolean;
  }
}

function configure<T extends Helpers>(opts: T) {} // TS1225
```

## Solutions

### Solution 1: Use the specific type from the namespace

```typescript
// Wrong
namespace MyTypes {
  export interface User {
    name: string;
    age: number;
  }
}

function greet(user: MyTypes): void {} // TS1225

// Correct
function greet(user: MyTypes.User): void {}
```

### Solution 2: Import the type directly

```typescript
// Wrong
import * as Models from "./models";
function process(item: Models): void {} // TS1225

// Correct
import type { Models } from "./models"; // if Models is an interface
// or
import { User } from "./models";
function process(item: User): void {}
```

### Solution 3: Reference the namespace member explicitly

```typescript
// Wrong
const settings: Config = { theme: "dark" }; // TS1225

// Correct
const settings: Config.Settings = { theme: "dark" };
```

## Examples

```typescript
namespace Validation {
  export interface StringValidator {
    isAcceptable(s: string): boolean;
  }

  export interface NumberValidator {
    isAcceptable(n: number): boolean;
  }
}

// Wrong — using namespace as type
function validate(v: Validation) {} // TS1225

// Correct — using specific interface from namespace
function validateString(v: Validation.StringValidator): boolean {
  return v.isAcceptable("hello");
}

function validateNumber(v: Validation.NumberValidator): boolean {
  return v.isAcceptable(42);
}
```
