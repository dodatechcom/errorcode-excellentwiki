# TS7029 — Fallthrough case in switch

TS7029 occurs when a `case` clause in a `switch` statement falls through to the next case without a `break`, `return`, or `throw` statement. TypeScript flags this to prevent accidental fallthrough bugs.

## Common Causes

```typescript
// Cause 1: Missing break in case
function getDayName(day: number): string {
  switch (day) {
    case 1:
      return "Monday";
    case 2:
      return "Tuesday";
    case 3: // TS7029 — falls through to case 4
    case 4: // TS7029 — falls through to default
    default:
      return "Weekend";
  }
}

// Cause 2: Accidental fallthrough
switch (command) {
  case "start":
    initialize();
    // missing break
  case "stop": // TS7029
    shutdown();
    break;
}

// Cause 3: Missing break after assignment
switch (x) {
  case 1:
    let result = "one"; // TS7029
  case 2:
    result = "two";
    break;
}
```

## Solutions

### Fix 1: Add break statements

```typescript
// Wrong
switch (x) {
  case 1:
    doSomething();
  case 2:
    doSomethingElse();
    break;
}

// Correct
switch (x) {
  case 1:
    doSomething();
    break;
  case 2:
    doSomethingElse();
    break;
}
```

### Fix 2: Explicitly mark intentional fallthrough

```typescript
switch (x) {
  case 1:
  // falls through intentionally
  case 2:
    handleOneOrTwo();
    break;
}
```

### Fix 3: Use return instead of break

```typescript
function handle(value: number): string {
  switch (value) {
    case 0:
      return "zero";
    case 1:
      return "one";
    default:
      return "other";
  }
}
```

### Fix 4: Return from the switch instead of breaking

```typescript
function getCategory(score: number): string {
  switch (true) {
    case score >= 90: return "A";
    case score >= 80: return "B";
    case score >= 70: return "C";
    default: return "F";
  }
}
```

## Examples

```typescript
// Example 1: Intentional fallthrough for grouped cases
function getDaysInMonth(month: number): number {
  switch (month) {
    case 1: case 3: case 5:
    case 7: case 8: case 10:
    case 12:
      return 31;
    case 4: case 6: case 9:
    case 11:
      return 30;
    case 2:
      return 28; // ignoring leap year
    default:
      throw new Error("Invalid month");
  }
}
```
