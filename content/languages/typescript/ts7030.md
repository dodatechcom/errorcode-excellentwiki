# TS7030 — Not all code paths return a value

TS7030 occurs when a function has a non-`void` return type but some code paths do not return a value. TypeScript checks that every possible execution path through a function ends with a `return` statement.

## Common Causes

```typescript
// Cause 1: Missing return in one branch of if/else
function isPositive(num: number): boolean {
  if (num > 0) {
    return true;
  }
  // TS7030 — missing return for num <= 0
}

// Cause 2: Missing default in switch
function getColorName(color: string): string {
  switch (color) {
    case "red":
      return "Red";
    case "blue":
      return "Blue";
  }
  // TS7030 — no default case
}

// Cause 3: Return only inside a conditional block
function process(value: number): number {
  if (value > 0) {
    return value * 2;
  }
  // TS7030 — no return for values <= 0
}
```

## Solutions

### Fix 1: Add the missing return

```typescript
// Wrong
function isPositive(num: number): boolean {
  if (num > 0) {
    return true;
  }
}

// Correct
function isPositive(num: number): boolean {
  if (num > 0) {
    return true;
  }
  return false;
}
```

### Fix 2: Add a default case to switch statements

```typescript
// Wrong
function getColorName(color: string): string {
  switch (color) {
    case "red": return "Red";
    case "blue": return "Blue";
  }
}

// Correct
function getColorName(color: string): string {
  switch (color) {
    case "red": return "Red";
    case "blue": return "Blue";
    default: return "Unknown";
  }
}
```

### Fix 3: Handle all branches

```typescript
// Wrong
function process(value: number): number {
  if (value > 0) {
    return value * 2;
  }
}

// Correct
function process(value: number): number {
  if (value > 0) {
    return value * 2;
  }
  return 0;
}
```

### Fix 4: Change return type to include undefined

```typescript
// If not all paths can logically return, allow undefined
function findUser(id: number): User | undefined {
  const user = users.get(id);
  if (user) {
    return user;
  }
  // return undefined is implied
}
```

## Examples

```typescript
// Example 1: Nested conditionals
function getGrade(score: number): string {
  if (score >= 90) return "A";
  if (score >= 80) return "B";
  if (score >= 70) return "C";
  if (score >= 60) return "D";
  return "F"; // final fallthrough
}
```
