TS2740 — Type ‘X’ is missing the following properties

TS2740 occurs when an object literal or value doesn’t have all the required properties that the target type demands. This is closely related to TS2741 but typically involves more complex type structures or when assigning to a variable rather than a function parameter.

Common Causes

// Cause 1: Missing properties in object literal assignment
interface User {
  name: string;
  age: number;
  email: string;
}
const user: User = { name: "Alice" }; // TS2740: missing 'age' and 'email'

// Cause 2: Missing properties from intersection type
interface A { a: string; }
interface B { b: number; }
type AB = A & B;
const ab: AB = { a: "hello" }; // TS2740: missing 'b'

// Cause 3: Class not implementing all interface members
interface Repository {
  find(id: number): any;
  save(entity: any): void;
  delete(id: number): void;
}
class SimpleRepo implements Repository {
  find(id: number) { return null; } // TS2740: missing 'save' and 'delete'
}

// Cause 4: Missing properties from generic constraint
interface Form<T> {
  values: T;
  errors: Record<keyof T, string>;
}
const form: Form<{ name: string }> = {
  values: { name: "" }
  // TS2740: missing 'errors'
};

Solutions

Fix 1: Add all missing properties

// Wrong
const user: User = { name: "Alice" };

// Correct
const user: User = { name: "Alice", age: 30, email: "alice@example.com" };

Fix 2: Use Partial if properties are optional

// Wrong
interface User {
  name: string;
  age: number;
}
const user: User = { name: "Alice" }; // TS2740

// Correct — use Partial for optional
const user: Partial<User> = { name: "Alice" };

Fix 3: Implement all interface members in classes

// Wrong
class SimpleRepo implements Repository {
  find(id: number) { return null; }
}

// Correct
class SimpleRepo implements Repository {
  find(id: number) { return null; }
  save(entity: any) { /* ... */ }
  delete(id: number) { /* ... */ }
}

Fix 4: Provide defaults or spread missing properties

const defaults = { age: 0, email: "" };
const user: User = { ...defaults, name: "Alice" };

Examples

// Example 1: React state type
interface State {
  count: number;
  loading: boolean;
  error: string | null;
}
this.setState({ count: 0 }); // TS2740: missing 'loading' and 'error'

// Fix: use Partial
this.setState({ count: 0 } as Partial<State>);

// Example 2: API response type
interface ApiResponse {
  status: number;
  data: unknown;
  headers: Record<string, string>;
}
const res: ApiResponse = { status: 200, data: {} }; // TS2740
