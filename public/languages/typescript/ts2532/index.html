TS2532 — Object is possibly ‘undefined’

TS2532 occurs when you access a property or call a method on a value that could be undefined. With strictNullChecks enabled, TypeScript requires you to verify that a value exists before using it.

Common Causes

// Cause 1: Array element access without bounds checking
const arr: number[] = [1, 2, 3];
const value = arr[10];
value.toFixed(); // TS2532: Object is possibly 'undefined'

// Cause 2: Object property access on optional property
interface Config {
  database?: { host: string };
}
function getHost(config: Config) {
  return config.database.host; // TS2532: database is possibly undefined
}

// Cause 3: Map.get() returns V | undefined
const map = new Map<string, number>();
const val = map.get("key");
val.toFixed(); // TS2532

// Cause 4: Function return type includes undefined
function findUser(id: number): User | undefined {
  return users.find(u => u.id === id);
}
const user = findUser(1);
console.log(user.name); // TS2532

Solutions

Fix 1: Use optional chaining

// Wrong
const val = map.get("key");
val.toFixed();

// Correct
const val = map.get("key");
val?.toFixed();

Fix 2: Add a truthiness check

// Wrong
const user = findUser(1);
console.log(user.name);

// Correct
const user = findUser(1);
if (user) {
  console.log(user.name);
}

Fix 3: Use default values with nullish coalescing

const val = map.get("key") ?? 0;
val.toFixed(); // OK — val is number

Fix 4: Use non-null assertion when certain

// Use ! only when you're sure the value exists
const val = map.get("key")!;
val.toFixed();

Examples

// Example 1: Nested optional properties
interface Company {
  ceo?: {
    name?: string;
  };
}
function getCEOName(company: Company) {
  return company.ceo.name; // TS2532: ceo and name are possibly undefined
}
// Fix: company.ceo?.name

// Example 2: Promise.then with possible undefined
Promise.resolve(42).then(result => {
  result.toFixed(); // OK — result is number
});
